问题场景 ​

作为Java后端开发，日常有很多重复性工作：批量修改数据、分析线上日志、给测试人员准备数据、从Excel导入配置。这些事情用Java写太重，手动操作又容易出错。Python是这类场景的最佳选择——几行代码就能搞定。

数据库批量操作 ​

场景：批量更新用户状态

产品说要把3个月没登录的用户标记为"休眠"，涉及10万条数据。手动在数据库里写SQL？万一改错了回不回来。

python
# batch_update_users.py
import pymysql
from datetime import datetime, timedelta

# 数据库连接配置
DB_CONFIG = {
    'host': '10.0.1.100',
    'port': 3306,
    'user': 'readonly_user',  # 先用readonly验证数量
    'password': 'xxx',
    'database': 'user_db',
    'charset': 'utf8mb4'
}

def count_inactive_users():
    """统计不活跃用户数量，先确认范围"""
    conn = pymysql.connect(**DB_CONFIG)
    try:
        with conn.cursor() as cursor:
            cutoff = datetime.now() - timedelta(days=90)
            sql = """
                SELECT COUNT(*) FROM t_user 
                WHERE last_login_time < %s AND status = 'ACTIVE'
            """
            cursor.execute(sql, (cutoff,))
            count = cursor.fetchone()[0]
            print(f"待更新用户数: {count}")
            return count
    finally:
        conn.close()

def batch_update(batch_size=1000):
    """分批更新，每批1000条，避免长事务"""
    config = DB_CONFIG.copy()
    config['user'] = 'write_user'  # 更新用写账号
    
    conn = pymysql.connect(**config)
    cutoff = datetime.now() - timedelta(days=90)
    
    try:
        total_updated = 0
        while True:
            with conn.cursor() as cursor:
                # 每次只更新batch_size条
                sql = """
                    UPDATE t_user 
                    SET status = 'DORMANT', updated_at = NOW() 
                    WHERE last_login_time < %s AND status = 'ACTIVE'
                    LIMIT %s
                """
                affected = cursor.execute(sql, (cutoff, batch_size))
                conn.commit()
                
                total_updated += affected
                print(f"已更新: {total_updated}, 本批: {affected}")
                
                if affected == 0:
                    break
        
        print(f"全部完成，共更新 {total_updated} 条")
    except Exception as e:
        conn.rollback()
        print(f"执行失败: {e}")
    finally:
        conn.close()

if __name__ == '__main__':
    count = count_inactive_users()
    if count > 0:
        confirm = input(f"确认更新 {count} 条数据？(y/n): ")
        if confirm == 'y':
            batch_update()

关键设计：

  • 先用只读账号统计数量，人工确认后再用写账号执行
  • 分批执行（LIMIT），避免一次更新10万行锁表
  • 每批单独commit，出错只回滚当前批次

日志分析 ​

场景：统计每个接口的QPS和错误率

python
# log_analyzer.py
import re
from collections import defaultdict
from datetime import datetime

# 日志格式: 2026-05-25 14:30:15.123 [http-nio-8080-exec-1] INFO OrderController - GET /api/orders/123 200 45ms
LOG_PATTERN = re.compile(
    r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}).*'
    r'(GET|POST|PUT|DELETE)\s+(/\S+)\s+(\d{3})\s+(\d+)ms'
)

def analyze_log(file_path):
    """分析日志文件，统计接口性能"""
    stats = defaultdict(lambda: {
        'count': 0, 'errors': 0, 'total_ms': 0, 'max_ms': 0
    })
    
    with open(file_path, 'r', encoding='utf-8') as f:
        for line in f:
            match = LOG_PATTERN.search(line)
            if match:
                time_str, method, path, status, cost_ms = match.groups()
                cost_ms = int(cost_ms)
                status = int(status)
                
                # 归一化路径（把ID参数替换为占位符）
                normalized_path = re.sub(r'/\d+', '/{id}', path)
                key = f"{method} {normalized_path}"
                
                stats[key]['count'] += 1
                stats[key]['total_ms'] += cost_ms
                if status >= 400:
                    stats[key]['errors'] += 1
                if cost_ms > stats[key]['max_ms']:
                    stats[key]['max_ms'] = cost_ms
    
    # 按调用次数排序输出
    print(f"{'接口':<40} {'调用次数':>8} {'错误数':>6} {'平均耗时':>8} {'最大耗时':>8} {'错误率':>8}")
    print("-" * 100)
    
    for path, data in sorted(stats.items(), key=lambda x: -x[1]['count']):
        avg_ms = data['total_ms'] / data['count'] if data['count'] > 0 else 0
        error_rate = data['errors'] / data['count'] * 100 if data['count'] > 0 else 0
        print(f"{path:<40} {data['count']:>8} {data['errors']:>6} {avg_ms:>7.0f}ms {data['max_ms']:>6}ms {error_rate:>7.2f}%")

if __name__ == '__main__':
    analyze_log('/data/logs/app/app.log')

API批量测试 ​

场景：迁移数据后验证接口返回是否正确

python
# api_checker.py
import requests
import json
from concurrent.futures import ThreadPoolExecutor, as_completed

BASE_URL = "http://10.0.1.100:8080"
HEADERS = {"Authorization": "Bearer your-token-here"}

def check_order(order_id):
    """检查单个订单的接口返回"""
    try:
        resp = requests.get(
            f"{BASE_URL}/api/orders/{order_id}",
            headers=HEADERS,
            timeout=5
        )
        if resp.status_code != 200:
            return (order_id, False, f"HTTP {resp.status_code}")
        
        data = resp.json()
        if data.get('code') != 200:
            return (order_id, False, data.get('message', '未知错误'))
        
        # 业务校验：检查关键字段是否存在
        order_data = data.get('data', {})
        required_fields = ['orderId', 'status', 'amount', 'items']
        missing = [f for f in required_fields if f not in order_data]
        if missing:
            return (order_id, False, f"缺少字段: {missing}")
        
        return (order_id, True, 'OK')
    except requests.Timeout:
        return (order_id, False, "请求超时")
    except Exception as e:
        return (order_id, False, str(e))

def batch_check(order_ids, max_workers=10):
    """并发检查多个订单"""
    results = []
    
    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        futures = {executor.submit(check_order, oid): oid for oid in order_ids}
        
        for future in as_completed(futures):
            result = future.result()
            results.append(result)
            
            if not result[1]:  # 失败的立即输出
                print(f"[FAIL] 订单 {result[0]}: {result[2]}")
    
    # 汇总
    success = sum(1 for r in results if r[1])
    fail = sum(1 for r in results if not r[1])
    print(f"\n检查完成: 成功 {success}, 失败 {fail}, 总计 {len(results)}")
    
    # 输出失败的订单
    if fail > 0:
        print("\n失败订单列表:")
        for r in results:
            if not r[1]:
                print(f"  订单 {r[0]}: {r[2]}")

if __name__ == '__main__':
    # 从文件读取订单ID列表
    with open('order_ids.txt', 'r') as f:
        order_ids = [line.strip() for line in f if line.strip()]
    
    batch_check(order_ids)

Excel数据处理 ​

场景：产品给了一份Excel，需要批量导入配置数据

python
# excel_importer.py
import openpyxl
import json

def read_excel_config(file_path):
    """读取Excel配置并转换为系统需要的格式"""
    wb = openpyxl.load_workbook(file_path)
    ws = wb.active
    
    # 第一行是表头
    headers = [cell.value for cell in ws[1]]
    print(f"表头: {headers}")
    
    configs = []
    for row in ws.iter_rows(min_row=2, values_only=True):
        if not row[0]:  # 跳过空行
            continue
        
        config = dict(zip(headers, row))
        # 类型转换
        if config.get('price'):
            config['price'] = float(config['price'])
        if config.get('stock'):
            config['stock'] = int(config['stock'])
        
        configs.append(config)
    
    print(f"读取到 {len(configs)} 条配置")
    return configs

def export_to_json(configs, output_path):
    """导出为JSON，方便检查"""
    with open(output_path, 'w', encoding='utf-8') as f:
        json.dump(configs, f, ensure_ascii=False, indent=2)
    print(f"已导出到 {output_path}")

def generate_sql(configs, table_name='t_product_config'):
    """生成INSERT SQL"""
    sqls = []
    for c in configs:
        sql = f"""INSERT INTO {table_name} (name, category, price, stock, status) 
VALUES ('{c['name']}', '{c['category']}', {c['price']}, {c['stock']}, 'ACTIVE');"""
        sqls.append(sql)
    
    output = '\n'.join(sqls)
    with open('import.sql', 'w', encoding='utf-8') as f:
        f.write(output)
    print(f"生成 {len(sqls)} 条SQL，保存到 import.sql")

if __name__ == '__main__':
    configs = read_excel_config('product_config.xlsx')
    export_to_json(configs, 'config_preview.json')
    generate_sql(configs)

定时报告 ​

场景：每天早上9点生成昨日数据报告并推送到钉钉群

python
# daily_report.py
import requests
import pymysql
from datetime import datetime, timedelta

DB_CONFIG = {
    'host': '10.0.1.100',
    'port': 3306,
    'user': 'report_user',
    'password': 'xxx',
    'database': 'order_db',
}

DINGTALK_WEBHOOK = "https://oapi.dingtalk.com/robot/send?access_token=your-token"

def get_yesterday_stats():
    """查询昨日业务数据"""
    yesterday = (datetime.now() - timedelta(days=1)).strftime('%Y-%m-%d')
    
    conn = pymysql.connect(**DB_CONFIG)
    try:
        with conn.cursor(pymysql.cursors.DictCursor) as cursor:
            # 新增订单数
            cursor.execute("""
                SELECT COUNT(*) as cnt, SUM(amount) as total 
                FROM t_order 
                WHERE DATE(create_time) = %s
            """, (yesterday,))
            order_stats = cursor.fetchone()
            
            # 新增用户数
            cursor.execute("""
                SELECT COUNT(*) as cnt 
                FROM t_user 
                WHERE DATE(create_time) = %s
            """, (yesterday,))
            user_stats = cursor.fetchone()
            
            return {
                'date': yesterday,
                'order_count': order_stats['cnt'],
                'order_amount': float(order_stats['total'] or 0),
                'new_users': user_stats['cnt']
            }
    finally:
        conn.close()

def send_to_dingtalk(stats):
    """发送到钉钉群"""
    message = {
        "msgtype": "markdown",
        "markdown": {
            "title": f"日报 - {stats['date']}",
            "text": f"""## 业务日报 {stats['date']}

| 指标 | 数值 |
|------|------|
| 新增订单 | {stats['order_count']} 笔 |
| 订单金额 | {stats['order_amount']:.2f} 元 |
| 新增用户 | {stats['new_users']} 人 |

> 自动生成，请勿回复"""
        }
    }
    
    resp = requests.post(DINGTALK_WEBHOOK, json=message, timeout=10)
    if resp.status_code == 200:
        print("报告发送成功")
    else:
        print(f"发送失败: {resp.text}")

if __name__ == '__main__':
    stats = get_yesterday_stats()
    send_to_dingtalk(stats)

踩坑总结 ​

  1. Python环境管理。用 pyenv 或 conda 管理Python版本，不要用系统自带的Python。pip install 的包要用 requirements.txt 锁版本
  2. 数据库操作一定要分批。一次性更新10万行可能锁表几分钟，分批执行（每批1000-5000条）对业务影响最小
  3. 脚本要做幂等设计。同一个脚本执行两次不能出问题。用状态字段判断已处理的记录，而不是用序号
  4. 敏感信息不要硬编码。数据库密码、API Token放在环境变量或配置文件里：
python
import os
DB_PASSWORD = os.environ.get('DB_PASSWORD')
  1. 日志用logging模块。不要到处print，用logging可以控制输出级别和格式：
python
import logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s %(levelname)s %(message)s')

Python不需要学得很深，够用就行。对于Java后端来说，它就是一个超级Shell，用来处理那些Java太重、Shell太弱的中间地带。建议团队维护一个公共脚本库，常见操作沉淀下来。
